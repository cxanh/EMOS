## 数据库设计

### Redis 数据结构

#### 1. 节点注册信息
```
Key: node:{node_id}:info
Type: Hash
Fields:
  - node_id: 节点ID
  - hostname: 主机名
  - ip: IP地址
  - registered_at: 注册时间
  - last_heartbeat: 最后心跳时间
TTL: 无 (永久保存)
```

#### 2. 最新监控数据
```
Key: node:{node_id}:latest
Type: Hash
Fields:
  - cpu_usage: CPU使用率
  - memory_usage: 内存使用率
  - disk_usage: 磁盘使用率
  - network_rx: 网络接收字节
  - network_tx: 网络发送字节
  - timestamp: 时间戳
TTL: 300s (5分钟)
```

#### 3. 在线节点集合
```
Key: nodes:online
Type: Set
Members: [node_id1, node_id2, ...]
TTL: 无
```

### InfluxDB 数据结构

#### Measurement: system_metrics

**Tags** (索引字段):
- node_id: 节点ID
- hostname: 主机名

**Fields** (数据字段):
- cpu_usage: float
- memory_usage: float
- disk_usage: float
- network_rx_bytes: int
- network_tx_bytes: int

**Timestamp**: 自动生成

**示例查询**:
```flux
from(bucket: "system_monitoring")
  |> range(start: -1h)
  |> filter(fn: (r) => r["_measurement"] == "system_metrics")
  |> filter(fn: (r) => r["node_id"] == "node001")
  |> filter(fn: (r) => r["_field"] == "cpu_usage")
```

## 认证授权机制

### JWT Token 结构

**Header**:
```json
{
  "alg": "HS256",
  "typ": "JWT"
}
```

**Payload**:
```json
{
  "user_id": "user001",
  "username": "admin",
  "role": "admin",
  "iat": 1707206400,
  "exp": 1707292800
}
```

**Secret**: 配置在环境变量 `JWT_SECRET`

### Token 使用

**HTTP 请求**:
```
Authorization: Bearer eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...
```

**WebSocket 连接**:
```
ws://localhost:8080/metrics?token=eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...
```

### 权限级别

- **admin**: 所有权限
- **user**: 只读权限
- **agent**: 仅上报数据权限

## 代码示例

### Python Agent 核心代码

```python
# agent.py
import psutil
import requests
import time
import yaml
from datetime import datetime

class SystemAgent:
    def __init__(self, config_file='config.yaml'):
        with open(config_file, 'r') as f:
            self.config = yaml.safe_load(f)
        
        self.node_id = self.config['node']['id']
        self.server_url = self.config['server']['url']
        self.api_endpoint = self.config['server']['api_endpoint']
        self.interval = self.config['collection']['interval']
    
    def collect_metrics(self):
        """采集系统指标"""
        return {
            'cpu_usage': psutil.cpu_percent(interval=1),
            'memory_usage': psutil.virtual_memory().percent,
            'disk_usage': psutil.disk_usage('/').percent,
            'network_rx_bytes': psutil.net_io_counters().bytes_recv,
            'network_tx_bytes': psutil.net_io_counters().bytes_sent
        }
    
    def send_metrics(self, metrics):
        """发送数据到中心服务"""
        data = {
            'node_id': self.node_id,
            'metrics': metrics,
            'timestamp': datetime.utcnow().isoformat() + 'Z'
        }
        
        try:
            response = requests.post(
                f"{self.server_url}{self.api_endpoint}",
                json=data,
                timeout=5
            )
            return response.status_code == 200
        except Exception as e:
            print(f"Error sending metrics: {e}")
            return False
    
    def run(self):
        """主循环"""
        print(f"Agent {self.node_id} started")
        while True:
            metrics = self.collect_metrics()
            success = self.send_metrics(metrics)
            if success:
                print(f"Metrics sent successfully")
            time.sleep(self.interval)

if __name__ == '__main__':
    agent = SystemAgent()
    agent.run()
```

### Node.js 后端核心代码

#### Redis 客户端封装

```javascript
// backend/config/redis.js
const redis = require('redis');

class RedisClient {
  constructor() {
    this.client = redis.createClient({
      host: process.env.REDIS_HOST || 'localhost',
      port: process.env.REDIS_PORT || 6379,
      password: process.env.REDIS_PASSWORD || undefined
    });
    
    this.client.on('error', (err) => {
      console.error('Redis error:', err);
    });
    
    this.client.connect();
  }
  
  async setLatestMetrics(nodeId, metrics) {
    const key = `node:${nodeId}:latest`;
    await this.client.hSet(key, metrics);
    await this.client.expire(key, 300); // 5分钟过期
  }
  
  async getLatestMetrics(nodeId) {
    const key = `node:${nodeId}:latest`;
    return await this.client.hGetAll(key);
  }
  
  async addOnlineNode(nodeId) {
    await this.client.sAdd('nodes:online', nodeId);
  }
  
  async getOnlineNodes() {
    return await this.client.sMembers('nodes:online');
  }
}

module.exports = new RedisClient();
```

#### Agent 接入路由

```javascript
// backend/routes/agent.js
const express = require('express');
const router = express.Router();
const redisClient = require('../config/redis');
const influxClient = require('../config/influxdb');
const { broadcastMetrics } = require('../websocket/metricsWS');

// Agent 注册
router.post('/register', async (req, res) => {
  try {
    const { node_id, hostname, ip } = req.body;
    
    // 保存节点信息到 Redis
    await redisClient.client.hSet(`node:${node_id}:info`, {
      node_id,
      hostname,
      ip,
      registered_at: new Date().toISOString()
    });
    
    // 添加到在线节点集合
    await redisClient.addOnlineNode(node_id);
    
    res.json({
      success: true,
      data: { node_id, registered: true }
    });
  } catch (error) {
    res.status(500).json({
      success: false,
      error: { message: error.message }
    });
  }
});

// 接收监控数据
router.post('/metrics', async (req, res) => {
  try {
    const { node_id, metrics, timestamp } = req.body;
    
    // 1. 保存到 Redis (最新数据)
    await redisClient.setLatestMetrics(node_id, {
      ...metrics,
      timestamp
    });
    
    // 2. 写入 InfluxDB (历史数据)
    await influxClient.writeMetrics(node_id, metrics, timestamp);
    
    // 3. 通过 WebSocket 推送给前端
    broadcastMetrics(node_id, { ...metrics, timestamp });
    
    res.json({
      success: true,
      message: '数据接收成功'
    });
  } catch (error) {
    res.status(500).json({
      success: false,
      error: { message: error.message }
    });
  }
});

module.exports = router;
```

import axios from 'axios';
import { useUserStore } from '@/stores/user';

const api = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL || 'http://localhost:3000/api',
  timeout: 10000

// 请求拦截器

  (config) => {
    const userStore = useUserStore();
    if (userStore.token) {
      config.headers.Authorization = `Bearer ${userStore.token}`;
    }
    return config;
  },
  (error) => {
    return Promise.reject(error);
  }
);

// 响应拦截器

  (response) => {
    return response.data;
  },
  (error) => {
    if (error.response?.status === 401) {
      const userStore = useUserStore();
      userStore.logout();
    }
    return Promise.reject(error);
  }
);

#### WebSocket 连接管理

```typescript
// frontend/src/utils/websocket.ts
import { useMetricsStore } from '@/stores/metrics';

class WebSocketManager {
  private ws: WebSocket | null = null;
  private reconnectTimer: number | null = null;
  private heartbeatTimer: number | null = null;
  
  connect(token: string) {
    const wsUrl = `ws://localhost:8080/metrics?token=${token}`;
    this.ws = new WebSocket(wsUrl);
    
    this.ws.onopen = () => {
      console.log('WebSocket connected');
      this.startHeartbeat();
    };
    
    this.ws.onmessage = (event) => {
      const data = JSON.parse(event.data);
      this.handleMessage(data);
    };
    
    this.ws.onclose = () => {
      console.log('WebSocket closed');
      this.stopHeartbeat();
      this.scheduleReconnect();
    };
    
    this.ws.onerror = (error) => {
      console.error('WebSocket error:', error);
    };
  }
  
  handleMessage(data: any) {
    const metricsStore = useMetricsStore();
    
    switch (data.type) {
      case 'metrics':
        metricsStore.updateMetrics(data.node_id, data.data);
        break;
      case 'pong':
        // 心跳响应
        break;
    }
  }
  
  subscribe(nodeId: string) {
    if (this.ws?.readyState === WebSocket.OPEN) {
      this.ws.send(JSON.stringify({
        type: 'subscribe',
        node_id: nodeId
      }));
    }
  }
  
  startHeartbeat() {
    this.heartbeatTimer = window.setInterval(() => {
      if (this.ws?.readyState === WebSocket.OPEN) {
        this.ws.send(JSON.stringify({ type: 'ping' }));
      }
    }, 30000); // 30秒
  }
  
  stopHeartbeat() {
    if (this.heartbeatTimer) {
      clearInterval(this.heartbeatTimer);
      this.heartbeatTimer = null;
    }
  }
  
  scheduleReconnect() {
    this.reconnectTimer = window.setTimeout(() => {
      console.log('Reconnecting...');
      this.connect(token);
    }, 5000); // 5秒后重连
  }
  
  disconnect() {
    this.stopHeartbeat();
    if (this.reconnectTimer) {
      clearTimeout(this.reconnectTimer);
    }
    this.ws?.close();
  }
}

export default new WebSocketManager();
```
